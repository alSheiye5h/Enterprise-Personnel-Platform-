table employe {
  code_employe varchar(20) // "HR-EMP-001" (Identifiant métier)
  cin varchar(20) // Numéro de sécurité sociale
  numero_passeport varchar(20) // Numéro de passeport (pour employés internationaux)

  // INFORMATIONS PERSONNELLES
  prenom varchar(50) // Prénom
  nom varchar(50) // Nom
  deuxieme_prenom varchar(30) // Deuxième prénom
  sexe char(1) // "M"/"F"/"O" (Genre)
  date_naissance date // struct tm from time.h - Date de naissance
  situation_matrimoniale varchar(15) // "Célibataire", "Marié", "Divorcé"
  nb_personnes_charge integer // Nombre de personnes à charge (pour calcul fiscal)

  // COORDONNÉES
  email_personnel varchar(100) // Email personnel
  email_professionnel varchar(100) // Email professionnel
  telephone_mobile varchar(20) // Téléphone mobile
  telephone_domicile varchar(20) // Téléphone domicile
  adresse_ligne1 varchar(100) // Adresse ligne 1
  adresse_ligne2 varchar(100) // Adresse ligne 2
  ville varchar(50) // Ville
  region varchar(50) // État/Région
  code_postal varchar(20) // Code postal
  pays varchar(50) // Pays

  // CONTACT D'URGENCE
  nom_contact_urgence varchar(100) // Nom du contact d'urgence
  telephone_contact_urgence varchar(20) // Téléphone du contact d'urgence
  relation_contact_urgence varchar(30) // Relation avec le contact

  // ATTRIBUTS MÉTIER (RELATIONS MCD)
  code_departement varchar(20) // Code département (Relation avec Département)
  code_poste varchar(20) // Code poste (Relation avec Poste)
  code_manager varchar(20) // Code manager (Relation réflexive)

  // STATUT PROFESSIONNEL
  statut_emploi varchar(20) // "Actif", "En congé", "Terminé"
  date_embauche date // Date d'embauche
  date_fin_periode_essai date // Date fin période d'essai
  date_depart date // Date de départ
  raison_depart varchar(50) // Raison du départ

  // CONFORMITÉ
  contrat_signe boolean // Contrat signé (0/1)
  verification_antecedents boolean // Vérification antécédents (0/1)
  verifie boolean // Employé vérifié (0/1)

  // RÈGLES MÉTIER
  est_valide boolean // Employé valide (attribut dérivé - règle métier)
}
