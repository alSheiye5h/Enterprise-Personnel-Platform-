table employe {
  code_employe varchar(20)        // identifiant interne comme HR-EMP-001
  cin varchar(20)                 // numéro CIN / sécurité sociale
  numero_passeport varchar(20)    // pour les internationaux
  
  // section infos personnelles
  prenom varchar(50)              // prénom principal
  nom varchar(50)                 // nom de famille
  deuxieme_prenom varchar(30)     // 2ème prénom si existe
  sexe char(1)                    // M/F/O
  date_naissance date             // date de naissance
  situation_matrimoniale varchar(15) // marié, célib, divorcé
  nb_personnes_charge integer     // pour calcul impôts
  
  // coordonnées diverses
  email_personnel varchar(100)    // perso
  email_professionnel varchar(100) // pro
  telephone_mobile varchar(20)    // portable
  telephone_domicile varchar(20)  // fixe maison
  adresse_ligne1 varchar(100)     // rue et numéro
  adresse_ligne2 varchar(100)     // complément adresse
  ville varchar(50)               // ville
  region varchar(50)              // région/état
  code_postal varchar(20)         // code postal
  pays varchar(50)                // pays
  
  // contact urgence
  nom_contact_urgence varchar(100)   // nom du contact
  telephone_contact_urgence varchar(20) // tel contact
  relation_contact_urgence varchar(30)  // lien familial
  
  // liens vers autres tables
  code_departement varchar(20)    // réf. département
  code_poste varchar(20)          // réf. poste
  code_manager varchar(20)        // son responsable
  
  // statut dans l'entreprise
  statut_emploi varchar(20)       // actif, congé, parti
  date_embauche date              // date d'entrée
  date_fin_periode_essai date     // fin période essai
  date_depart date                // date de sortie
  raison_depart varchar(50)       // motif départ
  
  // conformité administrative
  contrat_signe boolean           // oui/non
  verification_antecedents boolean // vérif faite ?
  verifie boolean                 // dossier complet
  
  // flag métier
  est_valide boolean              // actif et conforme
}

table departement {
  code_departement varchar(20)    // ex: DEP-IT, DEP-HR
  nom_departement varchar(100)    // nom complet
  type_departement varchar(30)    // fonctionnel, projet, etc.
  code_centre_cout varchar(20)    // code centre coût
  code_localisation varchar(20)   // lieu géo
  code_manager varchar(20)        // responsable dépt
  date_creation date              // date création
  budget_annuel decimal(12,2)     // budget annuel
  effectif_approuve integer       // headcount approuvé
  effectif_actuel integer         // taille réelle
  
  // hiérarchie
  code_departement_parent varchar(20) // dépt parent
  niveau_hierarchique integer     // niveau dans l'arbre
}

table poste {
  code_poste varchar(20)          // JOB-DEV-001
  intitule_poste varchar(100)     // intitulé officiel
  famille_metier varchar(50)      // technique, admin, etc.
  grade_poste varchar(10)         // grade G7, G8
  categorie_eeo varchar(10)       // classif EEO
  
  // fourchettes salariales
  salaire_minimum decimal(10,2)   // plancher
  salaire_median decimal(10,2)    // médian
  salaire_maximum decimal(10,2)   // plafond
  devise varchar(5)               // EUR, USD, etc.
  
  // prérequis
  annees_experience_requises integer // expérience requise
  niveau_education varchar(50)    // niveau études
  certifications_requises text    // certs obligatoires
}

table paie {
  id_paie varchar(30)             // PAY-2024-01-EMP001
  code_employe varchar(20)        // employé concerné
  periode_debut date              // début période
  periode_fin date                // fin période
  date_paiement date              // date versement
  
  // section revenus
  salaire_base decimal(10,2)      // salaire fixe
  allocation_logement decimal(10,2) // complément logement
  allocation_transport decimal(10,2) // transport
  heures_supplementaires decimal(10,2) // heures sup
  prime_performance decimal(10,2) // prime perf
  commission decimal(10,2)        // commissions
  autres_allocations decimal(10,2) // divers
  total_gains decimal(10,2)       // total brut
  
  // section retenues
  impot_revenu decimal(10,2)      // impôt sur revenu
  cotisation_cnss_employe decimal(10,2) // part salarié
  cotisation_cnss_employeur decimal(10,2) // part employeur
  assurance_sante_AMO decimal(10,2) // santé
  cotisation_retraite decimal(10,2) // retraite CIMR
  retenues_prets decimal(10,2)    // prélèvements prêts
  autres_deductions decimal(10,2) // autres
  total_deductions decimal(10,2)  // total retenues
  
  // net
  salaire_net decimal(10,2)       // net à payer
  
  // infos paiement
  mode_paiement varchar(20)       // virement, chèque
  numero_compte_bancaire varchar(30) // compte bancaire
  nom_banque varchar(50)          // banque
  statut_paiement varchar(20)     // payé, en attente
  
  // fiscal
  numero_certificat_fiscal varchar(30) // certif fiscal
  declaration_fiscale_deposee boolean  // déclaration faite
}

table pointage {
  id_pointage varchar(30)         // ATT-2024-01-01-EMP001
  code_employe varchar(20)        // employé
  date date                       // jour concerné
  
  // horaires
  heure_debut_prevue timestamp    // début prévu
  heure_fin_prevue timestamp      // fin prévu
  heure_debut_reelle timestamp    // début réel
  heure_fin_reelle timestamp      // fin réel
  
  // calculs temps
  heures_programmees decimal(4,2) // heures prévues
  heures_reelles decimal(4,2)     // heures faites
  heures_normales decimal(4,2)    // heures normales
  heures_supplementaires decimal(4,2) // heures sup
  heures_nuit decimal(4,2)        // nuit
  heures_jours_feries decimal(4,2) // fériés
  
  // statuts
  statut_presence varchar(20)     // présent, absent
  code_shift varchar(20)          // matin, soir, nuit
  statut_approbation varchar(20)  // approuvé ou non
  code_approbateur varchar(20)    // qui a approuvé
  
  remarques varchar(200)          // notes éventuelles
}

table conge {
  id_conge varchar(30)            // LVE-2024-EMP001-001
  code_employe varchar(20)        // demandeur
  type_conge varchar(30)          // annuel, maladie, etc.
  date_debut date                 // début congé
  date_fin date                   // fin congé
  total_jours decimal(5,2)        // nombre jours
  
  // workflow
  statut_demande varchar(20)      // brouillon, soumis...
  date_demande date               // quand demandé
  date_approbation date           // quand approuvé
  code_approbateur varchar(20)    // approbateur
  
  // gestion
  jours_deduits decimal(5,2)      // jours décomptés
  chemin_document_justificatif varchar(200) // justificatif
  est_medical boolean             // congé maladie ?
  numero_certificat_medical varchar(30) // no certif médical
  
  // lien paie
  est_conge_paye boolean          // payé ou non
}

table contrat {
  id_contrat varchar(30)          // CNT-EMP001-2024
  code_employe varchar(20)        // signataire
  type_contrat varchar(30)        // CDI, CDD, etc.
  date_debut date                 // début contrat
  date_fin date                   // fin si CDD
  date_signature date             // date signature
  
  // rémunération
  salaire_base decimal(10,2)      // salaire annuel/mensuel
  devise_salaire varchar(5)       // devise
  heures_par_semaine integer      // heures semaine
  periode_essai_jours integer     // jours période essai
  preavis_jours integer           // jours préavis
  
  // avantages
  assurance_sante boolean         // oui/non
  retraite boolean                // oui/non
  stock_options boolean           // oui/non
  jours_conges_annuels decimal(5,2) // congés annuels
  jours_conges_maladie decimal(5,2) // congés maladie
  
  // documents
  chemin_contrat varchar(200)     // chemin fichier
  chemin_copie_signee varchar(200) // copie signée
  signature_numerique boolean     // signé numérique
}

table configuration_fiscale {

  // tranches
  brackets json                   // tableau des tranches
  nombre_tranches integer         // nb de tranches
  
  // allocations
  allocation_personnelle decimal(10,2) // perso
  allocation_marie decimal(10,2)  // marié
  allocation_personne_charge decimal(10,2) // par charge
  
  // sécu sociale
  taux_employe decimal(5,4)       // taux salarié
  taux_employeur decimal(5,4)     // taux employeur
  plafond_cotisation decimal(10,2) // plafond
  
  // déclaration
  code_formulaire_fiscal varchar(20) // formulaire
  date_limite_declaration date    // deadline
}
